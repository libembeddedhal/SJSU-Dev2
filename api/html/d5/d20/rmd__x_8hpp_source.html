258  bool ResponseHandler(const Can::Message_t & message)
259  {
260  if (!(message.length == 8 && message.id == device_id_))
261  {
262  return false;
263  }
264 
265  switch (Commands(message.payload[0]))
266  {
267  case Commands::kReadMotorStatus2:
268  case Commands::kTorqueClosedLoopCommand:
269  case Commands::kSpeedClosedLoopCommand:
270  case Commands::kPositionClosedLoopCommand1:
271  case Commands::kPositionClosedLoopCommand2:
272  case Commands::kPositionClosedLoopCommand3:
273  case Commands::kPositionClosedLoopCommand4:
274  {
275  constexpr float kMax = 33.0f;
276  constexpr int16_t kMaxInt = BitLimits<16, int16_t>::Max();
277 
278  using units::angular_velocity::degrees_per_second_t;
279  using units::current::ampere_t;
280  using units::temperature::celsius_t;
281 
282  auto & payload = message.payload;
283 
284  auto temperature = static_cast<int8_t>(payload[1]);
285  auto current = static_cast<int16_t>((payload[3] << 8) | payload[2]);
286  auto speed = static_cast<int16_t>((payload[5] << 8) | payload[4]);
287  auto encoder = static_cast<int16_t>((payload[7] << 8) | payload[6]);
288 
289  auto temperature_float = static_cast<float>(temperature);
290  auto gear_ratio = CurrentSettings().gear_ratio;
291  auto speed_float = static_cast<float>(speed) / gear_ratio;
292  auto current_float = Map(current, -kMaxInt, kMaxInt, -kMax, kMax);
293 
294  feedback_.encoder_position = encoder;
295  feedback_.temperature = celsius_t{ temperature_float };
296  feedback_.current = ampere_t{ current_float };
297  feedback_.speed = degrees_per_second_t{ speed_float };
298 
299  break;
300  }
301  case Commands::kReadMotorStatus1AndErrorFlagCommands:
302  {
303  using units::temperature::celsius_t;
304  using units::voltage::volt_t;
305 
306  auto & payload = message.payload;
307 
308  uint8_t error = payload[7];
309  auto temperature = static_cast<int8_t>(payload[1]);
310  auto volts = static_cast<uint16_t>((payload[4] << 8) | payload[3]);
311 
312  auto temperature_float = static_cast<float>(temperature);
313  auto volts_float = static_cast<float>(volts);
314 
315  feedback_.temperature = celsius_t{ temperature_float };
316  feedback_.volts = volt_t{ volts_float * 0.1f };
317 
318  feedback_.over_voltage_protection_tripped = bit::Read(error, 0);
319  feedback_.over_temperature_protection_tripped = bit::Read(error, 3);
320 
321  break;
322  }
323  default: return false;
324  }
325 
326  return true;
327  }

329  void PollForMessages(uint8_t command_id,
330  std::chrono::nanoseconds receive_timeout)
331  {
332  sjsu::Can::Message_t response_message;
333  // Assume that the message has been missed. If the Wait() statement fails
334  // this will come true. If ResponseHandler() returns successful (true), then
335  // assign false to this variable.
336  feedback_.missed_feedback = true;
337 
338  // Poll/Wait for the CAN message retrieved from SecureGet() to return with
339  // payload contents where the first byte is equal to the expected
340  // command_id.
341  bool receive_successful =
342  Wait(receive_timeout, [this, command_id, &response_message]() {
343  response_message = node_->SecureGet();
344  if (response_message.payload[0] == command_id)
345  {
346  return true;
347  }
348  return false;
349  });
350 
351  LogDebug("CAN Message Contents:");
352  for (int i = 0; i < response_message.length; i++)
353  {
354  LogDebug(" 0x%02X", response_message.payload[i]);
355  }
356 
357  // Return early and keep feedback_.missed_feedback = true. This way the user
358  // knows if the previous feedback has been missed.
359  if (!receive_successful)
360  {
361  return;
362  }
363 
364  // Invert the response of ResponseHandler(). If it was successful in parsing
365  // feedback from the motor
366  feedback_.missed_feedback = !ResponseHandler(response_message);
367  }

369  int32_t ConvertRPMToCommandSpeed(
370  units::angular_velocity::revolutions_per_minute_t rpm,
371  float degree_per_second_ratio = 0.01f)
372  {
373  // Convert RPM to DPS (degrees per second)
374  units::angular_velocity::degrees_per_second_t dps = rpm;
375 
376  // Since the RPM is determined by the input motor speed, in order to make
377  // the output spin at that speed, the DSP must be scaled by the gear_ratio
378  // factor, otherwise for a gear ratio of 2:1, the output speed of the motor
379  // shaft will be half as fast. The RMD motors have a conversion of 0.01
380  // DPS/LSB meaning that the integer value of the scaled DPS must be x100 of
381  // the original value.
382  auto scaled_dps =
383  dps * (1.0f / degree_per_second_ratio) * CurrentSettings().gear_ratio;
384 
385  // Return the scaled DPS as an int32_t
386  return scaled_dps.to<int32_t>();
387  }

391  {
392  // Since the RPM is determined by the input motor speed, in order to make
393  // the output spin at that speed, the DSP must be scaled by the gear_ratio
394  // factor, otherwise for a gear ratio of 2:1, the output speed of the motor
395  // shaft will be half as fast. The RMD motors have a conversion of 0.01
396  // Degree/LSB meaning that the integer value of the scaled DPS must be x100
397  // of the original value.
398  auto scaled_angle =
399  angle * (1.0f / degree_to_int_ratio) * CurrentSettings().gear_ratio;

401  // Return the scaled DPS as an int32_t
