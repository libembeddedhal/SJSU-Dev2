51 static constexpr auto GenerateOversizedBaudTable() noexcept
52 {
53  constexpr uint32_t numerator_min = 1;
54  constexpr uint32_t denominator_min = 1;
55  constexpr uint32_t denominator_max = 15;
56 
57  TableIntermediate table{};
58 
59  // First element is the fractional 1.0f value. This allows baud rates that
60  // have dividers that are already integers, on the first iteration of the
61  // fractional value, this will match and the loop will stop.
62  table.element[table.valid_entries++] = UartFractionals{
63  .fraction = 1,
64  .byte_code = (1 << 4) | (0),
65  };
66 
67  // Skip zero numerator
68  for (uint32_t denominator = denominator_min; denominator < denominator_max;
69  denominator++)
70  {
71  for (uint32_t numerator = numerator_min; numerator < denominator;
72  numerator++)
73  {
74  float fraction = 1.0f + static_cast<float>(numerator) /
75  static_cast<float>(denominator);
76  if (1.1f <= fraction && fraction <= 1.9f)
77  {
78  table.element[table.valid_entries++] = UartFractionals{
79  .fraction = fraction,
80  .byte_code = static_cast<uint8_t>((denominator << 4) | (numerator)),
81  };
82  }
83  }
84  }
85 
86  return table;
87 }

197 constexpr static UartCalibration_t GenerateUartCalibration(
198  uint32_t baud_rate,
199  units::frequency::hertz_t peripheral_frequency)
200 {
201  uint32_t integer_peripheral_frequency =
202  units::unit_cast<uint32_t>(peripheral_frequency);
203  States state = States::kCalculateIntegerDivideLatch;
204  UartCalibration_t uart_calibration;
205  float baud_rate_float = static_cast<float>(baud_rate);
206  float divide_estimate = 0;
207  float decimal = kBaudTable[0].fraction;
208  size_t table_position = 0;
209  while (state != States::kDone)
210  {
211  switch (state)
212  {
213  case States::kCalculateIntegerDivideLatch:
214  {
215  divide_estimate =
216  DividerEstimate(baud_rate_float, 1, integer_peripheral_frequency);
217 
218  if (divide_estimate < 1.0f)
219  {
220  uart_calibration.divide_latch = 0;
221  state = States::kDone;
222  }
223  else if (IsDecimal(divide_estimate))
224  {
225  uart_calibration.divide_latch =
226  static_cast<uint32_t>(divide_estimate);
227  state = States::kDone;
228  }
229  else
230  {
231  state = States::kCalculateDivideLatchWithDecimal;
232  }
233  break;
234  }
235  case States::kCalculateDivideLatchWithDecimal:
236  {
237  divide_estimate = RoundFloat(DividerEstimate(
238  baud_rate_float, decimal, integer_peripheral_frequency));
239  decimal = FractionalEstimate(
240  baud_rate_float, divide_estimate, integer_peripheral_frequency);
241  if (1.1f <= decimal && decimal <= 1.9f)
242  {
243  state = States::kGenerateFractionFromDecimal;
244  }
245  else
246  {
247  state = States::kDecimalFailedGenerateNewDecimal;
248  }
249  break;
250  }
251  case States::kDecimalFailedGenerateNewDecimal:
252  {
253  if (table_position >= kBaudTable.size())
254  {
255  state = States::kDone;
256  }
257  decimal = kBaudTable[table_position++].fraction;
258  state = States::kCalculateDivideLatchWithDecimal;
259  break;
260  }
261  case States::kGenerateFractionFromDecimal:
262  {
263  uart_calibration = FindClosestFractional(decimal);
264  uart_calibration.divide_latch = static_cast<uint32_t>(divide_estimate);
265  state = States::kDone;
266  break;
267  }
268  case States::kDone:
269  default: break;
270  }
271  }
272  return uart_calibration;
273 }
