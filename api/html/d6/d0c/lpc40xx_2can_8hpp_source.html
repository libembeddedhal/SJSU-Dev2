323  void Send(const Message_t & message) override
324  {
325  LpcRegisters_t registers = ConvertMessageToRegisters(message);
326 
327  // Wait for one of the buffers to be free so we can transmit a message
328  // through it.
329  bool sent = false;
330  while (!sent)
331  {
332  uint32_t status_register = channel_.registers->SR;
333  // Check if any buffer is available.
334  if (bit::Read(status_register, BufferStatus::kTx1Released))
335  {
336  channel_.registers->TFI1 = registers.frame;
337  channel_.registers->TID1 = registers.id;
338  channel_.registers->TDA1 = registers.data_a;
339  channel_.registers->TDB1 = registers.data_b;
340  channel_.registers->CMR = Value(Commands::kSendTxBuffer1);
341  sent = true;
342  }
343  else if (bit::Read(status_register, BufferStatus::kTx2Released))
344  {
345  channel_.registers->TFI2 = registers.frame;
346  channel_.registers->TID2 = registers.id;
347  channel_.registers->TDA2 = registers.data_a;
348  channel_.registers->TDB2 = registers.data_b;
349  channel_.registers->CMR = Value(Commands::kSendTxBuffer2);
350  sent = true;

353  {

369  Message_t Receive() override
370  {
371  Message_t message;
372 
373  uint32_t frame = channel_.registers->RFS;
374 
375  // Extract all of the information from the message frame
376  bool is_remote_request = bit::Extract(frame, FrameInfo::kRemoteRequest);
377  uint32_t length = bit::Extract(frame, FrameInfo::kLength);
378  uint32_t format = bit::Extract(frame, FrameInfo::kFormat);
379 
380  message.is_remote_request = is_remote_request;
381  message.length = static_cast<uint8_t>(length);
382  message.format = static_cast<Message_t::Format>(format);
383 
384  // Get the frame ID
385  const uint32_t kRID = channel_.registers->RID;
386  if (message.format == Message_t::Format::kExtended)
387  {
388  message.id = bit::Extract(kRID, bit::MaskFromRange(0, 10));
389  }
390  else
391  {
392  message.id = bit::Extract(kRID, bit::MaskFromRange(0, 28));
393  }
394 
395  // Pull the bytes from RDA into the payload array
396  message.payload[0] = (channel_.registers->RDA >> (0 * 8)) & 0xFF;
397  message.payload[1] = (channel_.registers->RDA >> (1 * 8)) & 0xFF;
398  message.payload[2] = (channel_.registers->RDA >> (2 * 8)) & 0xFF;
399  message.payload[3] = (channel_.registers->RDA >> (3 * 8)) & 0xFF;
400 
401  // Pull the bytes from RDB into the payload array
402  message.payload[4] = (channel_.registers->RDB >> (0 * 8)) & 0xFF;
403  message.payload[5] = (channel_.registers->RDB >> (1 * 8)) & 0xFF;
404  message.payload[6] = (channel_.registers->RDB >> (2 * 8)) & 0xFF;
405  message.payload[7] = (channel_.registers->RDB >> (3 * 8)) & 0xFF;
406 
407  // Release the RX buffer and allow another buffer to be read.
408  channel_.registers->CMR = Value(Commands::kReleaseRxBuffer);
409 
410  return message;
411  }

413  bool SelfTest(uint32_t id) override
414  {
415  Message_t test_message;
416  test_message.id = id;
417 
418  // Enable reset mode in order to write to registers
419  SetMode(Mode::kReset, true);
420  // Enable self-test mode
421  SetMode(Mode::kSelfTest, true);
422  // Disable reset mode
423  SetMode(Mode::kReset, false);
424 
425  // Write test message to tx buffer 1
426  LpcRegisters_t registers = ConvertMessageToRegisters(test_message);
427 
428  // Set self-test request and send buffer 1
429  while (true)
430  {
431  uint32_t status_register = channel_.registers->SR;
432  // Check if any buffer is available.
433  if (bit::Read(status_register, BufferStatus::kTx1Released))
434  {
435  channel_.registers->TFI1 = registers.frame;
436  channel_.registers->TID1 = registers.id;
437  channel_.registers->TDA1 = 0;
438  channel_.registers->TDB1 = 0;
439  channel_.registers->CMR = Value(Commands::kSelfReceptionSendTxBuffer1);
440  break;
441  }
442  }
443 
444  // Allow time for RX to fire
445  Wait(100ms, [this]() { return HasData(); });
446 
447  // Read the message from the rx buffer and enqueue it into the rx queue.
448  auto received_message = Receive();
449 
450  // Check if the received message matches the one we sent
451  if (received_message.id != test_message.id)
452  {
453  return false;
454  }
455 
456  // Disable self-test mode
457  SetMode(Mode::kReset, true);
458  SetMode(Mode::kSelfTest, false);
459  SetMode(Mode::kReset, false);
460 
461  return true;
462  }

478  void ConfigureBaudRate()
479  {
480  // According to the BOSCH CAN spec, the nominal bit time is divided into 4
481  // time segments. These segments need to be programmed for the internal
482  // bit timing logic/state machine. Refer to the link below for more
483  // details about the importance of these time segments:
484  // http://www.keil.com/dd/docs/datashts/silabs/boschcan_ug.pdf
485  //
486  // Nominal Bit Time : 1 / 100 000 == 10^-5s
487  //
488  // 0_______________________________10^-5
489  // _/ 1 bit \_
490  // \_______________________________/
491  //
492  // | SYNC | PROP | PHASE1 | PHASE2 | BOSCH
493  // | TSCL | TSEG1 | TSEG2 | LPC
494  // ^
495  // sample point (industry standard: 80%)
496 
497  constexpr int kTseg1 = 0;
498  constexpr int kTseg2 = 0;
499  constexpr int kSyncJump = 0;
500  constexpr uint32_t kBaudRateAdjust = kTseg1 + kTseg2 + kSyncJump + 3;
501 
502  // Equation found p.563 of the user manual
503  // tSCL = CANsuppliedCLK * ((prescaler * kBaudRateAdjust) - 1)
504  // Configure the baud rate divider
505  auto & system = sjsu::SystemController::GetPlatformController();
506  const auto kFrequency = system.GetClockRate(channel_.id);
507  uint32_t prescaler =
508  (kFrequency / ((settings.baud_rate * kBaudRateAdjust)) - 1);
509 
510  sjsu::LogDebug(
511  "freq = %f :: prescale = %lu", kFrequency.to<double>(), prescaler);
512 
513  // Hold the results in RAM rather than altering the register directly
514  // multiple times.
515  bit::Value bus_timing;
516 
517  // Used to compensate for positive and negative edge phase errors. Defines
518  // how much the sample point can be shifted.
519  // These time segments determine the location of the "sample point".
520  bus_timing.Insert(0, BusTiming::kSyncJumpWidth)
521  .Insert(0, BusTiming::kTimeSegment1)
522  .Insert(0, BusTiming::kTimeSegment2)
523  .Insert(prescaler, BusTiming::kPrescalar);
524 
525  if (settings.baud_rate <= CanSettings_t::kStandardBaudRate)
526  {
527  // The bus is sampled 3 times (recommended for low speeds, 100kHz is
528  // considered HIGH).
529  bus_timing.Insert(1, BusTiming::kSampling);
530  }
531  else
532  {
533  bus_timing.Insert(0, BusTiming::kSampling);
534  }
535 
536  channel_.registers->BTR = bus_timing;
537  }

563  LpcRegisters_t ConvertMessageToRegisters(const Message_t & message) const
564  {
565  LpcRegisters_t registers;
566 
567  uint32_t frame_info =
568  bit::Value()
569  .Insert(message.length, FrameInfo::kLength)
570  .Insert(message.is_remote_request, FrameInfo::kRemoteRequest)
571  .Insert(Value(message.format), FrameInfo::kFormat);
572 
573  uint32_t data_a = 0;
574  data_a |= message.payload[0] << (0 * 8);
575  data_a |= message.payload[1] << (1 * 8);
576  data_a |= message.payload[2] << (2 * 8);
577  data_a |= message.payload[3] << (3 * 8);
578 
579  uint32_t data_b = 0;
580  data_b |= message.payload[4] << (0 * 8);
581  data_b |= message.payload[5] << (1 * 8);
582  data_b |= message.payload[6] << (2 * 8);
583  data_b |= message.payload[7] << (3 * 8);
584 
585  registers.frame = frame_info;
586  registers.id = message.id;
587  registers.data_a = data_a;
588  registers.data_b = data_b;
589 
590  return registers;
591  }
