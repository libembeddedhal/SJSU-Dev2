63  Acceleration_t Read() override
64  {
65  constexpr uint16_t kBytesPerAxis = 2;
66  constexpr uint8_t kNumberOfAxis = 3;
67 
68  Acceleration_t acceleration = {};
69 
70  uint8_t xyz_data[kBytesPerAxis * kNumberOfAxis];
71 
72  i2c_.WriteThenRead(kAddress,
73  { Value(RegisterMap::kXYZStartAddress) },
74  xyz_data,
75  sizeof(xyz_data));
76 
77  // First X-axis Byte (MSB first)
78  // =========================================================================
79  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
80  // XD15 | XD14 | XD13 | XD12 | XD11 | XD10 | XD9 | XD8
81  //
82  // Final X-axis Byte (LSB)
83  // =========================================================================
84  // Bit 7 | Bit 6 | Bit 5 | Bit 4 | Bit 3 | Bit 2 | Bit 1 | Bit 0
85  // XD7 | XD6 | XD5 | XD4 | XD3 | XD2 | XD1 | XD0
86  //
87  // We simply shift and OR the bytes together to get them into a signed int
88  // 16 value.
89 
90  int16_t x = static_cast<int16_t>(xyz_data[0] << 8 | xyz_data[1]);
91  int16_t y = static_cast<int16_t>(xyz_data[2] << 8 | xyz_data[3]);
92  int16_t z = static_cast<int16_t>(xyz_data[4] << 8 | xyz_data[5]);
93 
94  // Convert the 16 bit value into a floating point value m/S^2
95  constexpr int16_t kMax = std::numeric_limits<int16_t>::max();
96  constexpr int16_t kMin = std::numeric_limits<int16_t>::min();
97 
98  float x_ratio = sjsu::Map(x, kMin, kMax, -1.0f, 1.0f);
99  float y_ratio = sjsu::Map(y, kMin, kMax, -1.0f, 1.0f);
100  float z_ratio = sjsu::Map(z, kMin, kMax, -1.0f, 1.0f);
101 
102  acceleration.x = CurrentSettings().gravity * x_ratio;
103  acceleration.y = CurrentSettings().gravity * y_ratio;
104  acceleration.z = CurrentSettings().gravity * z_ratio;
105 
106  return acceleration;
107  }

110  void ConfigureFullScale()
111  {
112  uint32_t gravity_scale = settings.gravity.to<uint32_t>();
113  uint8_t gravity_code;
114 
115  switch (gravity_scale)
116  {
117  case 2: gravity_code = 0x00; break;
118  case 4: gravity_code = 0x01; break;
119  case 8: gravity_code = 0x02; break;
120  case 16: gravity_code = 0x03; break;
121  default:
122  throw Exception(std::errc::invalid_argument,
123  "Invalid gravity scale. Must be 2g, 4g, 8g or 16g.");
124  }
125 
126  // Write in the full scale range; but leave the self test and high pass
127  // filter config untouched
128  constexpr auto kScaleMask = bit::MaskFromRange(3, 4);
129  uint8_t config;
130  i2c_.WriteThenRead(
131  kAddress, { Value(RegisterMap::kDataConfig) }, &config, 1);
132 
133  config = bit::Insert(config, gravity_code, kScaleMask);
134 
135  i2c_.Write(kAddress, { Value(RegisterMap::kDataConfig), config });
136  }
