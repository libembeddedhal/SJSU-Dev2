344  void ModuleInitialize() override
345  {
347  auto & platform = sjsu::SystemController::GetPlatformController();
348  platform.PowerUpPeripheral(channel_.id); // checked
349 
351 
352  // Setup TX as output with alternative control (TXD)
353  channel_.td_pin.settings.function = 1;
354  channel_.td_pin.settings.Floating();
355 
356  // Setup RX as Input with Pullup
357  channel_.rd_pin.settings.function = 2; // Input Floating
358  channel_.rd_pin.settings.PullUp();
359 
360  channel_.td_pin.Initialize();
361  channel_.rd_pin.Initialize();

363  // Enter Initialization mode in order to write to CAN registers.
364  SetMasterMode(MasterControl::kSleepModeRequest, false);
365  SetMasterMode(MasterControl::kInitializationRequest, true);
366  SetMasterMode(MasterControl::kNoAutomaticRetransmission, true);
367  SetMasterMode(MasterControl::kAutomaticBussOffManagement, true);
368 
369  // Wait to enter Initialization mode
370  while (!GetMasterStatus(MasterStatus::kInitializationAcknowledge))
371  {
372  }

375  ConfigureReceiveHandler();
376 
377  EnableAcceptanceFilter();

379  // Leave Initialization mode
380  SetMasterMode(MasterControl::kInitializationRequest, false);
381 
382  // Wait to leave Initialization mode
383  while (GetMasterStatus(MasterStatus::kInitializationAcknowledge))
384  {
385  }

391  StmDataRegisters_t ConvertMessageToRegisters(const Message_t & message) const
392  {
393  StmDataRegisters_t registers;
394 
395  uint32_t frame_info =
396  bit::Value(0)
397  // .Insert(message.uptime, FrameLengthAndInfo::kMessageTimeStamp)
398  .Insert(message.length, FrameLengthAndInfo::kDataLengthCode);
399 
400  uint32_t frame_id = 0;
401 
402  if (message.format == Message_t::Format::kStandard)
403  {
404  frame_id =
405  bit::Value(0)
406  .Insert(true, MailboxIdentifier::kTransmitMailboxRequest)
407  .Insert(message.is_remote_request,
408  MailboxIdentifier::kRemoteRequest)
409  .Insert(Value(message.format), MailboxIdentifier::kIdentifierType)
410  .Insert(message.id, MailboxIdentifier::kStandardIdentifier);
411  }
412  else if (message.format == Message_t::Format::kExtended)
413  {
414  frame_id =
415  bit::Value(0)
416  .Insert(true, MailboxIdentifier::kTransmitMailboxRequest)
417  .Insert(message.is_remote_request,
418  MailboxIdentifier::kRemoteRequest)
419  .Insert(Value(message.format), MailboxIdentifier::kIdentifierType)
420  .Insert(message.id, MailboxIdentifier::kExtendedIdentifier);
421  }
422 
423  uint32_t data_a = 0;
424  data_a |= message.payload[0] << (0 * 8);
425  data_a |= message.payload[1] << (1 * 8);
426  data_a |= message.payload[2] << (2 * 8);
427  data_a |= message.payload[3] << (3 * 8);
428 
429  uint32_t data_b = 0;
430  data_b |= message.payload[4] << (0 * 8);
431  data_b |= message.payload[5] << (1 * 8);
432  data_b |= message.payload[6] << (2 * 8);
433  data_b |= message.payload[7] << (3 * 8);
434 
435  registers.frame = frame_info;
436  registers.id = frame_id;
437  registers.data_a = data_a;
438  registers.data_b = data_b;
439 
440  return registers;
441  }

443  void Send(const Message_t & message) override
444  {
445  StmDataRegisters_t registers = ConvertMessageToRegisters(message);
446  bool sent = false;
447  while (!sent)
448  {
449  uint32_t status_register = channel_.can->TSR;
450  // Check if any buffer is available.
451  if (bit::Read(status_register, TransmitStatus::kTransmitMailbox0Empty))
452  {
453  channel_.can->sTxMailBox[0].TDTR &= ~(0xF);
454  channel_.can->sTxMailBox[0].TDTR |= message.length & 0xF;
455  channel_.can->sTxMailBox[0].TDLR = registers.data_a;
456  channel_.can->sTxMailBox[0].TDHR = registers.data_b;
457  channel_.can->sTxMailBox[0].TIR = registers.id;
458  sent = true;
459  }
460  else if (bit::Read(status_register,
461  TransmitStatus::kTransmitMailbox1Empty))
462  {
463  channel_.can->sTxMailBox[1].TDTR &= ~(0xF);
464  channel_.can->sTxMailBox[1].TDTR |= message.length & 0xF;
465  channel_.can->sTxMailBox[1].TDLR = registers.data_a;
466  channel_.can->sTxMailBox[1].TDHR = registers.data_b;
467  channel_.can->sTxMailBox[1].TIR = registers.id;
468  sent = true;
469  }
470  else if (bit::Read(status_register,
471  TransmitStatus::kTransmitMailbox2Empty))
472  {
473  channel_.can->sTxMailBox[2].TDTR &= ~(0xF);
474  channel_.can->sTxMailBox[2].TDTR |= message.length & 0xF;
475  channel_.can->sTxMailBox[2].TDLR = registers.data_a;
476  channel_.can->sTxMailBox[2].TDHR = registers.data_b;
477  channel_.can->sTxMailBox[2].TIR = registers.id;
478  sent = true;
479  }
480  }

498  Message_t Receive() override
499  {
500  Message_t message;
501 
502  uint32_t fifo0_status = channel_.can->RF0R;
503  uint32_t fifo1_status = channel_.can->RF1R;
504  FIFOAssignment fifo_select = FIFOAssignment::kFIFONone;
505 
506  if (bit::Read(fifo0_status, FIFOStatus::kMessagesPending))
507  {
508  fifo_select = FIFOAssignment::kFIFO1;
509  }
510  else if (bit::Read(fifo1_status, FIFOStatus::kMessagesPending))
511  {
512  fifo_select = FIFOAssignment::kFIFO2;
513  }
514  else
515  {
516  // Error, tried to receive when there were no pending messages.
517  return message;
518  }
519 
520  uint32_t frame = channel_.can->sFIFOMailBox[Value(fifo_select)].RDTR;
521  uint32_t id = channel_.can->sFIFOMailBox[Value(fifo_select)].RIR;
522 
523  // Extract all of the information from the message frame
524  bool is_remote_request =
525  bit::Extract(id, MailboxIdentifier::kRemoteRequest);
526  uint32_t length = bit::Extract(frame, FrameLengthAndInfo::kDataLengthCode);
527  uint32_t format = bit::Extract(id, MailboxIdentifier::kIdentifierType);
528 
529  message.is_remote_request = is_remote_request;
530  message.length = static_cast<uint8_t>(length);
531  message.format = static_cast<Message_t::Format>(format);
532 
533  // Get the frame ID
534  if (message.format == Message_t::Format::kExtended)
535  {
536  message.id = bit::Extract(id, MailboxIdentifier::kExtendedIdentifier);
537  }
538  else
539  {
540  message.id = bit::Extract(id, MailboxIdentifier::kStandardIdentifier);
541  }
542 
543  // Pull the bytes from RDL into the payload array
544  message.payload[0] =
545  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDLR >> (0 * 8)) & 0xFF;
546  message.payload[1] =
547  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDLR >> (1 * 8)) & 0xFF;
548  message.payload[2] =
549  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDLR >> (2 * 8)) & 0xFF;
550  message.payload[3] =
551  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDLR >> (3 * 8)) & 0xFF;
552 
553  // Pull the bytes from RDH into the payload array
554  message.payload[4] =
555  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDHR >> (0 * 8)) & 0xFF;
556  message.payload[5] =
557  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDHR >> (1 * 8)) & 0xFF;
558  message.payload[6] =
559  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDHR >> (2 * 8)) & 0xFF;
560  message.payload[7] =
561  (channel_.can->sFIFOMailBox[Value(fifo_select)].RDHR >> (3 * 8)) & 0xFF;
562 
563  // Release the RX buffer and allow another buffer to be read.
564  if (fifo_select == FIFOAssignment::kFIFO1)
565  {
566  channel_.can->RF0R =
567  bit::Set(channel_.can->RF0R, FIFOStatus::kReleaseOutputMailbox);
568  }
569  else if (fifo_select == FIFOAssignment::kFIFO2)
570  {
571  channel_.can->RF1R =
572  bit::Set(channel_.can->RF1R, FIFOStatus::kReleaseOutputMailbox);
573  }
574 
575  return message;
576  }

579  {

581  test_message.id = id;
582 
583  // Enable self-test mode
584  DisableInterrupts();
585  SetLoopback();
586 
587  Send(test_message);
588 
589  // Allow time for RX to fire
590  Wait(100ms, [this]() { return HasData(); });
591 
592  // Read the message from the rx buffer and enqueue it into the rx queue.
593  auto received_message = Receive();
594 
595  // Check if the received message matches the one we sent
596  if (received_message.id != test_message.id)
597  {
598  return false;
599  }
600 
601  // Disable self-test mode
602  ClearLoopback();
603  EnableInterrupts();
604 
605  return true;

641  void ConfigureBaudRate()
642  {
643  constexpr int tseg1 = 4;
644  constexpr int tseg2 = 1;
645  constexpr int sync_jump = 0;
646 
647  constexpr uint32_t clocks_per_bit = tseg1 + tseg2 + sync_jump + 3;
648 
649  auto & system = sjsu::SystemController::GetPlatformController();
650  const auto frequency = system.GetClockRate(channel_.id);
651  const auto clock_rate = settings.baud_rate * clocks_per_bit;
652  uint32_t prescaler =
653  (frequency.to<uint32_t>() / (clock_rate.to<uint32_t>()) - 1);
654 
655  channel_.can->BTR =
656  bit::Insert(channel_.can->BTR, prescaler, BusTiming::kPrescalar);
657  channel_.can->BTR =
658  bit::Insert(channel_.can->BTR, tseg1, BusTiming::kTimeSegment1);
659  channel_.can->BTR =
660  bit::Insert(channel_.can->BTR, tseg2, BusTiming::kTimeSegment2);
661  channel_.can->BTR =
662  bit::Insert(channel_.can->BTR, sync_jump, BusTiming::kSyncJumpWidth);
