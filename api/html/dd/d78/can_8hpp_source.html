233  Can::Message_t SecureGet()
234  {
235  // Continuously check if the received CAN message is valid. NOTE: that, in
236  // general, the looping logic for this function almost never occurs as
237  // copies from RAM/cache are fast between ReceiveHandler and GetMessage()
238  while (true)
239  {
240  // First, atomically copy the access_counter to a variable.
241  const auto kReadMessageStart = access_counter.load();
242 
243  // Copy the contents of the mesage into the kCanMessage variable.
244  const Can::Message_t kCanMessage = data;
245 
246  // Finally, atomically copy the access_counter again such that the value
247  // between the start and finish can be compared.
248  const auto kReadMessageFinish = access_counter.load();
249 
250  // If and only if these two values are the same, meaning that the
251  // ReceiveHandler isn't currently modifying the Node_t's data,
252  // can the message be considered valid.
253  if (kReadMessageStart == kReadMessageFinish)
254  {
255  return kCanMessage;
256  }
257  }
258 
259  return {};
260  };
261 
262  private:
263  friend CanNetwork;
264 
269  void Update(const Can::Message_t & new_data)
270  {
271  // Atomic increment of the access counter to notify any threads that are
272  // using GetMesage() that the value of this node is changing.
273  access_counter++;
274 
275  // Copy the contents of the new message into the map.
276  data = new_data;
277 
278  // Another atomic increment of the access counter to notify any thread
279  // using GetMessage() that the value of this node has finished changing.
280  // For GetMessage to be valid, the function must save the access_counter,
281  // copy the message data, then check the access_counter again to make sure
282  // it has not changed. If it has changed, then GetMessage() should
283  // continue polling the map node until access_counter is the same between
284  // copies.
285  access_counter++;
286  }

325  [[nodiscard]] Node_t * CaptureMessage(uint32_t id)
326  {
327  try
328  {
329  Node_t empty_node;
330 
331  // Insert an empty Node_t into the map with the following ID as
332  // the key. This is how each ID is remembered when calls to
333  // ReceiveHandler() and GetMessage() are performed. ReceiveHandler() will

377  void ReceiveHandler(sjsu::Can & can)
378  {
379  // If there isn't any data available, return early.
380  if (!can.HasData())
381  {
382  return;
383  }
384 
385  // Pop the latest can message off the queue.
386  const auto kMessage = can.Receive();
387 
388  // Check if the map already has a value for this ID. This acts as the last
389  // stage of the CAN filter for the CANBUS Network module. If the key
390  // does NOT exist in the map, then this message will not be saved.
391  // Typically, this only happens when the hardware filter has not been setup
392  // properly to eliminate can messages that should be ignored.
393  //
394  // Map lookups can be costly, especially in a interrupt context, so only
395  // needing to hash/lookup the ID once is preferred. To prevent multiple
396  // lookups, an iterator is stored into the message_node variable.
397  auto message_node = messages_.find(kMessage.id);
398 
399  // If the ID has an associated value in the map, then the node std::pair<>
400  // is returned, otherwise, the ::end() node is returned.
401  if (message_node != messages_.end())
402  {
403  message_node->second.Update(kMessage);
404  }
405  }
