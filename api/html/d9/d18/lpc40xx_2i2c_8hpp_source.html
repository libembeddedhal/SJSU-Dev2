90  static void I2cHandler(const Port_t & i2c)
91  {
92  MasterState state = MasterState(i2c.registers->STAT);
93  uint32_t clear_mask = 0;
94  uint32_t set_mask = 0;
95  switch (state)
96  {
97  case MasterState::kBusError: // 0x00
98  {
99  i2c.transaction.status = std::errc::io_error;
100  set_mask = Control::kAssertAcknowledge | Control::kStop;
101  break;
102  }
103  case MasterState::kStartCondition: // 0x08
104  {
105  i2c.registers->DAT = i2c.transaction.GetProperAddress();
106  break;
107  }
108  case MasterState::kRepeatedStart: // 0x10
109  {
110  i2c.transaction.operation = Operation::kRead;
111  i2c.registers->DAT = i2c.transaction.GetProperAddress();
112  break;
113  }
114  case MasterState::kSlaveAddressWriteSentReceivedAck: // 0x18
115  {
116  clear_mask = Control::kStart;
117  if (i2c.transaction.out_length == 0)
118  {
119  i2c.transaction.busy = false;
120  set_mask = Control::kStop;
121  }
122  else
123  {
124  size_t position = i2c.transaction.position++;
125  i2c.registers->DAT = i2c.transaction.data_out[position];
126  }
127  break;
128  }
129  case MasterState::kSlaveAddressWriteSentReceivedNack: // 0x20
130  {
131  clear_mask = Control::kStart;
132  i2c.transaction.busy = false;
133  i2c.transaction.status = std::errc::no_such_device_or_address;
134  set_mask = Control::kStop;
135  break;
136  }
137  case MasterState::kTransmittedDataReceivedAck: // 0x28
138  {
139  if (i2c.transaction.position >= i2c.transaction.out_length)
140  {
141  if (i2c.transaction.repeated)
142  {
143  // OR with 1 to set address as READ for the next transaction
144  i2c.transaction.operation = Operation::kRead;
145  i2c.transaction.position = 0;
146  set_mask = Control::kStart;
147  }
148  else
149  {
150  i2c.transaction.busy = false;
151  set_mask = Control::kStop;
152  }
153  }
154  else
155  {
156  size_t position = i2c.transaction.position++;
157  i2c.registers->DAT = i2c.transaction.data_out[position];
158  }
159  break;
160  }
161  case MasterState::kTransmittedDataReceivedNack: // 0x30
162  {
163  i2c.transaction.busy = false;
164  set_mask = Control::kStop;
165  break;
166  }
167  case MasterState::kArbitrationLost: // 0x38
168  {
169  set_mask = Control::kStart;
170  break;
171  }
172  case MasterState::kSlaveAddressReadSentReceivedAck: // 0x40
173  {
174  clear_mask = Control::kStart;
175  if (i2c.transaction.in_length == 0)
176  {
177  set_mask = Control::kStop;
178  }
179  // If we only want 1 byte, make sure to nack that byte
180  else if (i2c.transaction.in_length == 1)
181  {
182  clear_mask |= Control::kAssertAcknowledge;
183  }
184  // If we want more then 1 byte, make sure to ack the first byte
185  else
186  {
187  set_mask = Control::kAssertAcknowledge;
188  }
189  break;
190  }
191  case MasterState::kSlaveAddressReadSentReceivedNack: // 0x48
192  {
193  clear_mask = Control::kStart;
194  i2c.transaction.status = std::errc::no_such_device_or_address;
195  i2c.transaction.busy = false;
196  set_mask = Control::kStop;
197  break;
198  }
199  case MasterState::kReceivedDataReceivedAck: // 0x50
200  {
201  const size_t kBufferEnd = i2c.transaction.in_length;
202  if (i2c.transaction.position < kBufferEnd)
203  {
204  const size_t kPosition = i2c.transaction.position;
205  i2c.transaction.data_in[kPosition] =
206  static_cast<uint8_t>(i2c.registers->DAT);
207  i2c.transaction.position++;
208  }
209  // Check if the position has been pushed past the buffer length
210  if (i2c.transaction.position + 1 >= kBufferEnd)
211  {
212  clear_mask = Control::kAssertAcknowledge;
213  i2c.transaction.busy = false;
214  }
215  else
216  {
217  set_mask = Control::kAssertAcknowledge;
218  }
219  break;
220  }
221  case MasterState::kReceivedDataReceivedNack: // 0x58
222  {
223  i2c.transaction.busy = false;
224  if (i2c.transaction.in_length != 0)
225  {
226  size_t position = i2c.transaction.position++;
227  i2c.transaction.data_in[position] =
228  static_cast<uint8_t>(i2c.registers->DAT);
229  }
230  set_mask = Control::kStop;
231  break;
232  }
233  case MasterState::kDoNothing: // 0xF8
234  {
235  break;
236  }
237  default:
238  {
239  clear_mask = Control::kStop;
240  SJ2_ASSERT_FATAL(false, "Invalid I2C State Reached!!");
241  break;
242  }
243  }
244  // Clear I2C Interrupt flag
245  clear_mask |= Control::kInterrupt;
246  // Set register controls
247  i2c.registers->CONSET = set_mask;
248  i2c.registers->CONCLR = clear_mask;
249  }

327  void ConfigureClockRate()
328  {
329  // Calculating and setting the I2C Clock rate
330  // Weight the high side duty cycle more than the lower side by 30% in
331  // order to give more time for the bus to charge up.
332  const auto kPeripheralFrequency =
333  sjsu::SystemController::GetPlatformController().GetClockRate(i2c_.id);
334 
335  const float kClockDivider = kPeripheralFrequency / settings.frequency;
336  const float kScll = kClockDivider * settings.duty_cycle;
337  const float kSclh = kClockDivider * (1 - settings.duty_cycle);
338 
339  i2c_.registers->SCLL = static_cast<uint32_t>(kScll);
340  i2c_.registers->SCLH = static_cast<uint32_t>(kSclh);
341  }

346  void BlockUntilFinished() const
347  {
348  if (!IsEnabled())
349  {
350  throw Exception(
351  std::errc::operation_not_permitted,
352  "Attempt to use I2C, before peripheral was not INITIALIZED! Be sure "
353  "to run the i2c.Initialize() method first");
354  }
355 
356  auto wait_for_i2c_transaction = [this]() -> bool {
357  return !i2c_.transaction.busy;
358  };
359 
360  auto wait_status = Wait(i2c_.transaction.timeout, wait_for_i2c_transaction);
361 
362  if (i2c_.transaction.status == CommonErrors::kBusError)
363  {
364  throw CommonErrors::kBusError;
365  }
366  else if (i2c_.transaction.status == CommonErrors::kDeviceNotFound)
367  {
368  throw CommonErrors::kDeviceNotFound;
369  }
370  else if (!wait_status)
371  {
372  // Abort I2C communication if this point is reached!
373  i2c_.registers->CONSET = Control::kAssertAcknowledge | Control::kStop;
374 
375  if (i2c_.transaction.out_length == 0 || i2c_.transaction.in_length == 0)
376  {
377  throw CommonErrors::kTimeout;
378  }
379  }
380 
381  // Ensure that start is cleared before leaving this function, to end the
382  // transaction.
383  i2c_.registers->CONCLR = Control::kStart;
384  }
